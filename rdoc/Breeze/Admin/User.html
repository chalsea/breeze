    # File app/models/breeze/admin/user.rb, line 77
def self.roles
  @_roles ||= ({}).tap do |hash|
    ROLES.each do |role|
      hash[role] = I18n::t role, :scope => [ :breeze, :users, :roles ], :default => role.to_s.humanize
    end
  end
end
